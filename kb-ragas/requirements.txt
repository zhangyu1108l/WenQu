# Python dependencies for the Ragas evaluation sidecar.
# All versions are locked for reproducible Docker builds and stable runtime behavior.

# Web framework for exposing the evaluation API; 0.111.0 is a stable FastAPI release with mature Pydantic v2 support.
fastapi==0.111.0

# ASGI server with standard extras for production-friendly HTTP serving; 0.29.0 works reliably with FastAPI 0.111.0.
uvicorn[standard]==0.29.0

# Core RAG evaluation framework. Strictly lock ragas because its APIs change quickly; 0.1.9 avoids upgrade breakage.
ragas==0.1.9

# LLM interface layer required by Ragas. Strictly lock LangChain because Ragas 0.1.9 depends on the 0.1.x API shape.
langchain==0.1.20

# OpenAI-compatible client layer for connecting DeepSeek through base_url; 0.1.7 matches LangChain 0.1.20.
langchain-openai==0.1.7

# Community integrations used by LangChain/Ragas 0.1.x; 0.0.38 is the compatible companion release.
langchain-community==0.0.38

# Dataset format expected by ragas.evaluate(); 2.19.0 is stable with ragas 0.1.9.
datasets==2.19.0

# Data processing and metric aggregation; 2.2.2 is a stable release for the current Python runtime line.
pandas==2.2.2

# Synchronous HTTP callback client for posting results back to the Java service; 2.32.0 keeps callback behavior stable.
requests==2.32.0

# Loads local .env configuration in development and containers; 1.0.1 is a stable, minimal runtime dependency.
python-dotenv==1.0.1

# Async HTTP client for non-blocking service calls; 0.27.0 aligns with FastAPI tooling in this stack.
httpx==0.27.0
