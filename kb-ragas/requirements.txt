# Ragas 评估侧车的 Python 依赖。
# 所有版本均锁定，以保证 Docker 构建可复现且运行时行为稳定。

# 用于暴露评估 API 的 Web 框架；0.111.0 是稳定版本，并成熟支持 Pydantic v2。
fastapi==0.111.0

# 带标准扩展的 ASGI 服务器，适合生产 HTTP 服务；0.29.0 与 FastAPI 0.111.0 搭配可靠。
uvicorn[standard]==0.29.0

# 核心 RAG 评估框架；0.4.3 支持本地开发使用的 Python 3.13 运行时。
ragas==0.4.3

# Ragas 所需的 LLM 接口层；1.3.4 在 Python 3.13 和 NumPy 2.x 下依赖解析稳定。
langchain==1.3.4

# OpenAI 兼容客户端层，用于通过 base_url 连接 DeepSeek。
langchain-openai==1.2.2

# Ragas 使用的社区集成；0.3.31 保留 Ragas 导入的 vertexai 兼容模块，
# 同时仍支持兼容 Python 3.13 的 NumPy 2.x wheel。
langchain-community==0.3.31

# ragas.evaluate() 期望的数据集格式。
datasets==5.0.0

# 数据处理和指标聚合；2.3.3 提供 Python 3.13 的 Windows wheel。
pandas==2.3.3

# Ragas、pandas、datasets 和 LangChain 使用的数值运行时。
# 锁定 NumPy 2.x，避免 Python 3.13 在 Windows 上尝试从源码构建不受支持的 NumPy 1.x。
numpy==2.4.6

# 同步 HTTP 回调客户端，用于把结果回传给 Java 服务。
requests==2.32.5

# 在开发环境和容器中加载本地 .env 配置；1.0.1 是稳定且轻量的运行时依赖。
python-dotenv==1.0.1

# 异步 HTTP 客户端，用于非阻塞服务调用；0.27.0 与本技术栈中的 FastAPI 工具链匹配。
httpx==0.27.0
