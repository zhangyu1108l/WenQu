# Python dependencies for the Ragas evaluation sidecar.
# All versions are locked for reproducible Docker builds and stable runtime behavior.

# Web framework for exposing the evaluation API; 0.111.0 is a stable FastAPI release with mature Pydantic v2 support.
fastapi==0.111.0

# ASGI server with standard extras for production-friendly HTTP serving; 0.29.0 works reliably with FastAPI 0.111.0.
uvicorn[standard]==0.29.0

# Core RAG evaluation framework. 0.4.3 supports the Python 3.13 runtime used in local development.
ragas==0.4.3

# LLM interface layer required by Ragas. 1.3.4 resolves cleanly on Python 3.13 with NumPy 2.x.
langchain==1.3.4

# OpenAI-compatible client layer for connecting DeepSeek through base_url.
langchain-openai==1.2.2

# Community integrations used by Ragas. 0.3.31 keeps the vertexai compatibility module Ragas imports,
# while still supporting Python 3.13-compatible NumPy 2.x wheels.
langchain-community==0.3.31

# Dataset format expected by ragas.evaluate().
datasets==5.0.0

# Data processing and metric aggregation; 2.3.3 provides Python 3.13 Windows wheels.
pandas==2.3.3

# Numeric runtime used by Ragas, pandas, datasets, and LangChain. Pin NumPy 2.x to avoid Python 3.13
# attempting to build unsupported NumPy 1.x from source on Windows.
numpy==2.4.6

# Synchronous HTTP callback client for posting results back to the Java service.
requests==2.32.5

# Loads local .env configuration in development and containers; 1.0.1 is a stable, minimal runtime dependency.
python-dotenv==1.0.1

# Async HTTP client for non-blocking service calls; 0.27.0 aligns with FastAPI tooling in this stack.
httpx==0.27.0
